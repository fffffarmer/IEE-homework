	$(function(){
		
		// 校验是否包含特殊字符
		function validateStr(str) {
			var arrs = new Array("<script", "</script", "<iframe", "</iframe", "<frame", "</frame", "set-cookie", 
				"%3cscript", "%3c/script", "%3ciframe", "%3c/iframe", "%3cframe", "%3c/frame", "javascript:", "<body", "</body",
				"%3cbody", "%3c/body", "<img", "%3cimg", "%");
				
			for (i = 0; i < arrs.length; i ++) {
				if (str.indexOf(arrs[i]) > -1) {
					return true;
				}
			}
			return false;
		}
		
		$(document).on('blur','#studentQuestionForm input[name="email"]',function(e){
			var ele = $(this);
			var email = ele.val();
			if(email && email.trim().length < 1){
				ele.parent().parent().addClass('error');
				$('#studentQuestionForm em.email').html('Email required');
				return ;
			}else{
				// 校验邮箱规则
				var Regex = /^(?:\w+\.?)*\w+@(?:\w+\.)*\w+$/;
				if(Regex.test(email) == false){
					ele.parent().parent().addClass('error');
					$('#studentQuestionForm em.email').html('Email address is invalid.');
					return ;
				}
			}
			ele.parent().parent().removeClass('error');
			$('#studentQuestionForm em.email').html('');
		});
		$(document).on('blur','#studentQuestionForm textarea[name="question"]',function(e){
			var ele = $(this);
			var password = ele.val();
			if(password.trim().length < 1){
				ele.parent().parent().addClass('error');
				$('#studentQuestionForm em.question').html('Message requried.');
				return ;
			}
			if(validateStr(password)){
				ele.parent().parent().addClass('error');
				$('#studentQuestionForm em.question').html('Message illegal.');
				return ;
			}
			ele.parent().parent().removeClass('error');
			$('#studentQuestionForm em.question').html('');
		});
		
		$(document).on('submit','#studentQuestionForm',function(e){

			if($('#studentQuestionForm input[name="email"]').val() == ''){

				$('#studentQuestionForm em.question').html('Message requried.');

			}

			$.ajax({
				type : 'POST',
				url : action, // 请求的action路径
				data : ele.serialize(),
				error : function() { // 请求失败处理函数
